CREATE OR REPLACE FUNCTION unique_surat_masuk_bkd()
  RETURNS trigger AS
$BODY$
DECLARE
temprowid integer;
tempstatus CHARACTER VARYING(1);
tempinfo TEXT;
tempquery TEXT;
BEGIN
	
	--IF TG_OP = 'INSERT' OR (TG_OP = 'UPDATE' AND (COALESCE(NULLIF(NEW.STATUS, ''), NULL) IS NULL OR NEW.STATUS = '2')) THEN
	IF TG_OP = 'INSERT' OR TG_OP = 'UPDATE' THEN
		IF TG_OP = 'INSERT' THEN
			IF COALESCE(NULLIF(NEW.NO_AGENDA, ''), NULL) IS NULL THEN
			temprowid:= 0;
			ELSE
				SELECT
				CASE WHEN COUNT(1) > 0 THEN 1 ELSE 0 END INTO temprowid
				FROM persuratan.surat_masuk_bkd A
				WHERE 1=1
				AND EXISTS
				(
					SELECT 1
					FROM
					(
						SELECT A.NO_AGENDA
						FROM PERSURATAN.SURAT_MASUK_BKD A
						INNER JOIN PERSURATAN.SURAT_MASUK_BKD_DISPOSISI B ON A.SURAT_MASUK_BKD_ID = B.SURAT_MASUK_BKD_ID
						WHERE SURAT_AWAL = 1 
						AND TO_CHAR(NOW(), 'YYYY') = TO_CHAR(A.TANGGAL, 'YYYY')
					) X
					WHERE A.NO_AGENDA = X.NO_AGENDA
				)
				AND A.NO_AGENDA = NEW.NO_AGENDA
				;
			END IF;

			raise notice 'sql %',NEW.SURAT_MASUK_BKD_ID || '-' || temprowid || '-' || NEW.NO_AGENDA || '-' || TO_CHAR(NEW.TANGGAL, 'YYYY');
			--RETURN FALSE; 
		ELSE
		
			IF COALESCE(NULLIF(NEW.NO_AGENDA, ''), NULL) IS NULL THEN
				temprowid:= NULL;
			ELSE
				SELECT
				SURAT_MASUK_BKD_ID INTO temprowid
				FROM persuratan.surat_masuk_bkd A
				WHERE 1=1
				AND EXISTS
				(
					SELECT 1
					FROM
					(
						SELECT A.NO_AGENDA, A.SURAT_MASUK_BKD_ID
						FROM PERSURATAN.SURAT_MASUK_BKD A
						INNER JOIN PERSURATAN.SURAT_MASUK_BKD_DISPOSISI B ON A.SURAT_MASUK_BKD_ID = B.SURAT_MASUK_BKD_ID
						WHERE SURAT_AWAL = 1 
						AND TO_CHAR(NOW(), 'YYYY') = TO_CHAR(A.TANGGAL, 'YYYY')
					) X
					WHERE A.NO_AGENDA = X.NO_AGENDA
					AND A.SURAT_MASUK_BKD_ID = X.SURAT_MASUK_BKD_ID
				)
				AND A.NO_AGENDA = NEW.NO_AGENDA
				;
			END IF;

			raise notice 'sql %',NEW.SURAT_MASUK_BKD_ID || '-' || temprowid || '-' || NEW.NO_AGENDA || '-' || TO_CHAR(NEW.TANGGAL, 'YYYY');

			IF NEW.SURAT_MASUK_BKD_ID = temprowid OR temprowid IS NULL THEN
			temprowid:= 0;
			ELSE
			temprowid:= 1;
			END IF;
			
		END IF;
		--raise notice 'sql %',NEW.surat_masuk_bkd_ID;
		IF temprowid >= 1 AND temprowid IS NOT NULL THEN
			RETURN FALSE; 
		ELSE
			RETURN NEW;
		END IF;
	ELSE
	RETURN NEW;
	END IF;
	
END;
$BODY$
LANGUAGE plpgsql VOLATILE COST 100;
ALTER FUNCTION unique_surat_masuk_bkd() OWNER TO postgres;

			SELECT SATUAN_KERJA_ASAL_ID INTO tempsatuankerjaid FROM PERSURATAN.SURAT_MASUK_BKD_DISPOSISI WHERE SURAT_MASUK_BKD_ID = NEW.SURAT_MASUK_BKD_ID AND STATUS_POSISI_SURAT = 1;
			tempinfo:= 'Dapat di ambil di ';
		
		--usul nomor ke surat keluar : STATUS_SURAT_KELUAR in 1, 2
		ELSIF NEW.STATUS_BERKAS = '10' AND NEW.STATUS_VERIFIKASI = 1 AND (NEW.JENIS_ID = 3 OR NEW.JENIS_ID = 4 OR NEW.JENIS_ID = 7 OR NEW.JENIS_ID = 10) AND NEW.STATUS_SURAT_KELUAR IN (1,2) --AND NEW.USULAN_SURAT_ID = 1 
		THEN
			--raise notice 'sql %','A';
			RETURN NEW;
			
		ELSIF NEW.STATUS_SURAT_KELUAR IS NOT NULL OR (OLD.STATUS_VERIFIKASI = 1 AND NEW.STATUS_VERIFIKASI = 1) OR NEW.STATUS_SURAT_KELUAR = 2 
		THEN
			--raise notice 'sql %',NEW.JENIS_ID || '-B-' || NEW.USULAN_SURAT_ID;
			RETURN NEW;
		ELSIF NEW.STATUS_BERKAS = '10' AND OLD.STATUS_VERIFIKASI IS NULL AND NEW.STATUS_VERIFIKASI = 1 AND (NEW.JENIS_ID = 3 OR NEW.JENIS_ID = 4 OR NEW.JENIS_ID = 7 OR NEW.JENIS_ID = 10) THEN
			SELECT SATUAN_KERJA_ASAL_ID INTO tempsatuankerjaid FROM PERSURATAN.SURAT_MASUK_BKD_DISPOSISI WHERE SURAT_MASUK_BKD_ID = NEW.SURAT_MASUK_BKD_ID AND STATUS_POSISI_SURAT = 1;
			tempinfo:= 'Terverifikasi oleh ';
		ELSIF NEW.STATUS_BERKAS = '10' AND OLD.STATUS_VERIFIKASI IS NULL AND NEW.STATUS_VERIFIKASI = 1 THEN
			SELECT SATUAN_KERJA_ASAL_ID INTO tempsatuankerjaid FROM PERSURATAN.SURAT_MASUK_BKD_DISPOSISI WHERE SURAT_MASUK_BKD_ID = NEW.SURAT_MASUK_BKD_ID AND STATUS_POSISI_SURAT = 1;
			tempinfo:= 'Terverifikasi dan proses pengesahan oleh ';
		-- kalau turun status kembali 1 return
		ELSIF NEW.STATUS_BERKAS = '10' AND OLD.STATUS_KEMBALI = 1 AND NEW.STATUS_KEMBALI = 1 THEN
		RETURN NEW;
		ELSIF NEW.STATUS_BERKAS = '10' AND OLD.STATUS_KEMBALI IS NULL AND NEW.STATUS_KEMBALI = 1 THEN
			SELECT SATUAN_KERJA_ASAL_ID INTO tempsatuankerjaid FROM PERSURATAN.SURAT_MASUK_BKD_DISPOSISI WHERE SURAT_MASUK_BKD_ID = NEW.SURAT_MASUK_BKD_ID AND STATUS_POSISI_SURAT = 1;
			tempinfo:= 'Turun Status oleh ';
		ELSIF NEW.STATUS_BERKAS = '10' AND OLD.STATUS_TMS IS NULL AND NEW.STATUS_TMS = 1 THEN
			SELECT SATUAN_KERJA_ASAL_ID INTO tempsatuankerjaid FROM PERSURATAN.SURAT_MASUK_BKD_DISPOSISI WHERE SURAT_MASUK_BKD_ID = NEW.SURAT_MASUK_BKD_ID AND STATUS_POSISI_SURAT = 1;
			tempinfo:= 'TMS oleh ';
		ELSIF NEW.STATUS_BERKAS = '10' AND OLD.STATUS_TMS IS NOT NULL AND NEW.STATUS_TMS IS NULL THEN
			SELECT SATUAN_KERJA_ASAL_ID INTO tempsatuankerjaid FROM PERSURATAN.SURAT_MASUK_BKD_DISPOSISI WHERE SURAT_MASUK_BKD_ID = NEW.SURAT_MASUK_BKD_ID AND STATUS_POSISI_SURAT = 1;
			tempinfo:= 'Batal TMS oleh ';
		ELSIF NEW.STATUS_BERKAS = '6' OR NEW.STATUS_BERKAS = '8' OR NEW.STATUS_BERKAS = '10' THEN
			SELECT SATUAN_KERJA_ASAL_ID INTO tempsatuankerjaid FROM PERSURATAN.SURAT_MASUK_BKD_DISPOSISI WHERE SURAT_MASUK_BKD_ID = NEW.SURAT_MASUK_BKD_ID AND STATUS_POSISI_SURAT = 1;
			tempinfo:= 'Di terima oleh ';
		ELSIF NEW.STATUS_BERKAS = '7' OR NEW.STATUS_BERKAS = '9' THEN
			SELECT SATUAN_KERJA_TUJUAN_ID INTO tempsatuankerjaid FROM PERSURATAN.SURAT_MASUK_BKD_DISPOSISI WHERE SURAT_MASUK_BKD_ID = NEW.SURAT_MASUK_BKD_ID AND STATUS_POSISI_SURAT = 1;
			tempinfo:= 'Disposisi ke ';
		END IF;
	--tempinfo:= tempinfo || 'Tempat: ' || REPLACEQUOTE(NEW.TEMPAT) || '<br/>';
	--END IF;
	raise notice 'sql %',tempinfo;

	IF NEW.STATUS_BERKAS = '1' OR NEW.STATUS_BERKAS = '2' OR NEW.STATUS_BERKAS = '3' THEN
		IF NEW.USER_LOGIN_PEGAWAI_ID IS NULL THEN
			tempquery:= '
			INSERT INTO persuratan.SURAT_MASUK_PEGAWAI_LOG(SURAT_MASUK_PEGAWAI_ID, JENIS_ID, SURAT_MASUK_UPT_ID, PEGAWAI_ID, SATUAN_KERJA_ID, INFO_LOG, LAST_USER, LAST_DATE, USER_LOGIN_ID, USER_LOGIN_PEGAWAI_ID)
			VALUES(
			'|| NEW.SURAT_MASUK_PEGAWAI_ID ||',
			'|| NEW.JENIS_ID ||',
			'|| NEW.SURAT_MASUK_UPT_ID ||',
			'|| NEW.PEGAWAI_ID ||',
			'|| tempsatuankerjaid ||',
			'''|| tempinfo ||''',
			'''|| NEW.LAST_USER ||''',
			'''|| NOW() ||'''
			, '|| NEW.USER_LOGIN_ID ||'
			, NULL
			)
			';
		ELSE
			tempquery:= '
			INSERT INTO persuratan.SURAT_MASUK_PEGAWAI_LOG(SURAT_MASUK_PEGAWAI_ID, JENIS_ID, SURAT_MASUK_UPT_ID, PEGAWAI_ID, SATUAN_KERJA_ID, INFO_LOG, LAST_USER, LAST_DATE, USER_LOGIN_ID, USER_LOGIN_PEGAWAI_ID)
			VALUES(
			'|| NEW.SURAT_MASUK_PEGAWAI_ID ||',
			'|| NEW.JENIS_ID ||',
			'|| NEW.SURAT_MASUK_UPT_ID ||',
			'|| NEW.PEGAWAI_ID ||',
			'|| tempsatuankerjaid ||',
			'''|| tempinfo ||''',
			'''|| NEW.LAST_USER ||''',
			'''|| NOW() ||'''
			, '|| NEW.USER_LOGIN_ID ||'
			, '|| NEW.USER_LOGIN_PEGAWAI_ID ||'
			)
			';
		END IF;
	ELSIF NEW.STATUS_BERKAS = '4' OR NEW.STATUS_BERKAS = '5' OR NEW.STATUS_BERKAS = '6' OR NEW.STATUS_BERKAS = '7' OR NEW.STATUS_BERKAS = '8' OR NEW.STATUS_BERKAS = '9' OR NEW.STATUS_BERKAS = '10' THEN
		IF NEW.USER_LOGIN_PEGAWAI_ID IS NULL THEN
			tempquery:= '
			INSERT INTO persuratan.SURAT_MASUK_PEGAWAI_LOG(SURAT_MASUK_PEGAWAI_ID, JENIS_ID, SURAT_MASUK_BKD_ID, PEGAWAI_ID, SATUAN_KERJA_ID, INFO_LOG, LAST_USER, LAST_DATE, USER_LOGIN_ID, USER_LOGIN_PEGAWAI_ID)
			VALUES(
			'|| NEW.SURAT_MASUK_PEGAWAI_ID ||',
			'|| NEW.JENIS_ID ||',
			'|| NEW.SURAT_MASUK_BKD_ID ||',
			'|| NEW.PEGAWAI_ID ||',
			'|| tempsatuankerjaid ||',
			'''|| tempinfo ||''',
			'''|| NEW.LAST_USER ||''',
			'''|| NOW() ||'''
			, '|| NEW.USER_LOGIN_ID ||'
			, NULL
			)
			';
		ELSE
			tempquery:= '
			INSERT INTO persuratan.SURAT_MASUK_PEGAWAI_LOG(SURAT_MASUK_PEGAWAI_ID, JENIS_ID, SURAT_MASUK_BKD_ID, PEGAWAI_ID, SATUAN_KERJA_ID, INFO_LOG, LAST_USER, LAST_DATE, USER_LOGIN_ID, USER_LOGIN_PEGAWAI_ID)
			VALUES(
			'|| NEW.SURAT_MASUK_PEGAWAI_ID ||',
			'|| NEW.JENIS_ID ||',
			'|| NEW.SURAT_MASUK_BKD_ID ||',
			'|| NEW.PEGAWAI_ID ||',
			'|| tempsatuankerjaid ||',
			'''|| tempinfo ||''',
			'''|| NEW.LAST_USER ||''',
			'''|| NOW() ||'''
			, '|| NEW.USER_LOGIN_ID ||'
			, '|| NEW.USER_LOGIN_PEGAWAI_ID ||'
			)
			';
		END IF;
	END IF;
	
	raise notice 'sql %',tempquery;
	IF tempquery IS NOT NULL THEN         
	EXECUTE tempquery;
	END IF;

	--khusus pembatalan
	IF NEW.STATUS_BERKAS = '99' THEN
		tempinfo:= 'Pembatalan oleh ';
		INSERT INTO persuratan.SURAT_MASUK_PEGAWAI_LOG
		(
			SURAT_MASUK_PEGAWAI_ID, JENIS_ID
			, SURAT_MASUK_UPT_ID, SURAT_MASUK_BKD_ID, PEGAWAI_ID, INFO_LOG
			, LAST_USER, LAST_DATE, USER_LOGIN_ID, USER_LOGIN_PEGAWAI_ID
		)
		VALUES
		(
			NEW.SURAT_MASUK_PEGAWAI_ID
			, NEW.JENIS_ID
			, NEW.SURAT_MASUK_UPT_ID
			, NEW.SURAT_MASUK_BKD_ID
			, NEW.PEGAWAI_ID
			, tempinfo
			, NEW.LAST_USER
			, NOW()
			, NEW.USER_LOGIN_ID
			, NEW.USER_LOGIN_PEGAWAI_ID
		);
	END IF;
	
	RETURN NEW;
   
END;
